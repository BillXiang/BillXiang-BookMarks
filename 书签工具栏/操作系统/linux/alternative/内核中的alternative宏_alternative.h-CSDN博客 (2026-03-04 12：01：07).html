X86 和 arm 等体系结构实现了指令动态替换的功能，即 alternative 机制。在该机制的作用下，内核可以根据当前cpu是否支持某些软硬件feature来实现对内核代码的在线优化，即在不关机、不换内核的情况下在线改写某些内核指令，以达到加速内核执行的目的。下面以 x86 为例描述其大概的实现过程。
1）alternative() 宏定义

在 arch/x86/include/asm/alternative-asm.h 定义了 ALTERNATIVE() 宏，如下所示：

  1. .macro ALTERNATIVE oldinstr, newinstr, feature
  2. 140:
  3.     \oldinstr
  4. 141:
  5.     .skip -(((144f-143f)-(141b-140b)) > 0) * ((144f-143f)-(141b-140b)),0x90
  6. 142:
  7.     
  8.     .pushsection .altinstructions,"a"
  9.     altinstruction_entry 140b,143f,\feature,142b-140b,144f-143f,142b-141b
  10.     .popsection
  11.     
  12.     .pushsection .altinstr_replacement,"ax"
  13. 143:
  14.     \newinstr
  15. 144:
  16.     .popsection
  17. .endm
cpp
运行

由于是根据 feature 改写指令，因此这里需要有 oldinstr 和 newinstr，还有 feature
ALTERNATIVE() 宏主要做了以下两项工作：
a、将一个 struct alt_instr 实例放入 vmlinux ELF 文件的 .altinstructions section，该 结构体 存储了指令改写所依赖的必要信息，如下所示：

  1. struct alt_instr {
  2.     s32 instr_offset;   /* original instruction */
  3.     s32 repl_offset;    /* offset to replacement instruction */
  4.     u16 cpuid;      /* cpuid bit set for replacement */
  5.     u8  instrlen;       /* length of original instruction */
  6.     u8  replacementlen; /* length of new instruction */
  7.     u8  padlen;     /* length of build-time padding */
  8. } __packed;
cpp
运行

b、将上面的 newinstr 放入 .altinstr_replacement section

2）alternative() 宏调用

定义 alternative_input() 宏

  1. #define alternative_input(oldinstr, newinstr, feature, input...)    \
  2.     asm_inline volatile (ALTERNATIVE(oldinstr, newinstr, feature)   \
  3.         : : "i" (0), ## input)
cpp
运行

在 ./arch/x86/include/asm/processor.h 文件调用 alternative_input()

alternative_input(BASE_PREFETCH, "prefetchnta %P1", X86_FEATURE_XMM, "m" (*(const char *)x));
cpp
运行

3）在线改写指令
在 arch/x86/kernel/alternative.c 文件中调用了 apply_alternatives()函数，它实现了指令的在线改写：

apply_alternatives(__alt_instructions, __alt_instructions_end);
cpp
运行

其中，__alt_instructions 和 __alt_instructions_end 分别是 .altinstructions section 的起始和结束位置，它们的引用方式是：

extern struct alt_instr __alt_instructions[], __alt_instructions_end[];
cpp
运行

它们定义在 arch/x86/kernel/vmlinux.lds.S，如下所示：

  1. /*  
  2.  * struct alt_inst entries. From the header (alternative.h):
  3.  * "Alternative instructions for different CPU types or capabilities"
  4.  * Think locking instructions on spinlocks.
  5.  */
  6.  . = ALIGN(8);
  7.  .altinstructions : AT(ADDR(.altinstructions) - LOAD_OFFSET) {
  8.      __alt_instructions = .;
  9.      *(.altinstructions)
  10.      __alt_instructions_end = .;
  11.   }
cpp
运行

apply_alternatives() 的定义如下，大体流程是遍历 .altinstructions section 中的所有 struct alt_instr 实例，分别对其进行处理：

  1. void __init_or_module noinline apply_alternatives(struct alt_instr *start,
  2.                           struct alt_instr *end)
  3. {
  4.     struct alt_instr *a;
  5.     u8 *instr, *replacement;
  6.     u8 insn_buff[MAX_PATCH_LEN];
  7.  
  8.     DPRINTK("alt table %px, -> %px", start, end);
  9.     /*   
  10.      * The scan order should be from start to end. A later scanned
  11.      * alternative code can overwrite previously scanned alternative code.
  12.      * Some kernel functions (e.g. memcpy, memset, etc) use this order to
  13.      * patch code.
  14.      *
  15.      * So be careful if you want to change the scan order to any other
  16.      * order.
  17.      */
  18.     for (a = start; a < end; a++) {
  19.         int insn_buff_sz = 0; 
  20.  
  21.         instr = (u8 *)&a->instr_offset + a->instr_offset;
  22.         replacement = (u8 *)&a->repl_offset + a->repl_offset;
  23.         BUG_ON(a->instrlen > sizeof(insn_buff));
  24.         BUG_ON(a->cpuid >= (NCAPINTS + NBUGINTS) * 32); 
  25.         if (!boot_cpu_has(a->cpuid)) {
  26.             if (a->padlen > 1) 
  27.                 optimize_nops(a, instr);
  28.  
  29.             continue;
  30.         }    
  31.  
  32.         DPRINTK("feat: %d*32+%d, old: (%pS (%px) len: %d), repl: (%px, len: %d), pad: %d",
  33.             a->cpuid >> 5,
  34.             a->cpuid & 0x1f,
  35.             instr, instr, a->instrlen,
  36.             replacement, a->replacementlen, a->padlen);
  37.  
  38.         DUMP_BYTES(instr, a->instrlen, "%px: old_insn: ", instr);
  39.         DUMP_BYTES(replacement, a->replacementlen, "%px: rpl_insn: ", replacement);
  40.  
  41.         memcpy(insn_buff, replacement, a->replacementlen);
  42.         insn_buff_sz = a->replacementlen;
  43.         
  44.         /*   
  45.          * 0xe8 is a relative jump; fix the offset.
  46.          *   
  47.          * Instruction length is checked before the opcode to avoid
  48.          * accessing uninitialized bytes for zero-length replacements.
  49.          */  
  50.         if (a->replacementlen == 5 && *insn_buff == 0xe8) {
  51.             *(s32 *)(insn_buff + 1) += replacement - instr;
  52.             DPRINTK("Fix CALL offset: 0x%x, CALL 0x%lx",
  53.                 *(s32 *)(insn_buff + 1),
  54.                 (unsigned long)instr + *(s32 *)(insn_buff + 1) + 5);
  55.         }    
  56.              
  57.         if (a->replacementlen && is_jmp(replacement[0]))
  58.             recompute_jump(a, instr, replacement, insn_buff);
  59.              
  60.         if (a->instrlen > a->replacementlen) {
  61.             add_nops(insn_buff + a->replacementlen,
  62.                  a->instrlen - a->replacementlen);
  63.             insn_buff_sz += a->instrlen - a->replacementlen;
  64.         }    
  65.         DUMP_BYTES(insn_buff, insn_buff_sz, "%px: final_insn: ", instr);
  66.              
  67.         text_poke_early(instr, insn_buff, insn_buff_sz);
  68.     }        
  69. }            
cpp
运行
