RISC-V 异常处理在 KVM 中的实现

1 前言

Trap 处理是 RISC-V 虚拟化实现中的重要部分，包括异常和中断两个部分。当前 KVM 是 RISC-V 虚拟化扩展在软件层面较为可靠的实现，本文将结合 RISC-V 特权指令集手册的规定，分析 KVM 中有关异常处理的实现，中断部分由于涉及较多驱动层面的内容，故将在之后的文章中结合 MMIO，timer 等做具体探讨。

3 KVM 异常处理

3.1 异常处理入口

在 KVM 对 RISC-V H 扩展的实现中，与异常处理相关的函数调用关系如下图所示。目前的实现中，KVM 能够处理三类异常。即虚拟机内的 page fault、虚拟指令异常和系统调用，三种不同的异常处理分别对应了不同的实现。

3.2 异常分类及其定义

kvm_arch_vcpu_ioctl_run 函数用于实现 vCPU 的运行。其调用 kvm_riscv_vcpu_enter_exit 函数进入 vCPU 的运行，此时 Guest 进入运行状态，CPU 处于 VS 或者 VU 模式。当 Guest 发生无法处理的异常时，Guest 退出，CPU 进入 HS 模式，随后 KVM 调用 kvm_riscv_vcpu_exit 来实现对异常的处理。

kvm_riscv_vcpu_exit 函数内部包含三个部分，分别对应三种异常的处理，代码如下：

  1. // arch/riscv/kvm/vcpu_exit.c: line 188
  2. /*
  3.  * Return > 0 to return to guest, < 0 on error, 0 (and set exit_reason) on
  4.  * proper exit to userspace.
  5.  */
  6. int kvm_riscv_vcpu_exit(struct kvm_vcpu *vcpu, struct kvm_run *run,
  7. 			struct kvm_cpu_trap *trap)
  8. {
  9. 	int ret;
  11. 	/* 仅处理 guest 内部的 trap */
  12. 	/* If we got host interrupt then do nothing */
  13. 	if (trap->scause & CAUSE_IRQ_FLAG)
  14. 		return 1;
  16. 	/* 处理 guest 中断：KVM 使得 Linux 内核成为了 hypervisor，故 KVM 内部实现了对来自 guest/VM 和 hypervisor 的 trap 处理，此处仅处理 Guest 内部的 trap */
  17. 	/* Handle guest traps */
  18. 	ret = -EFAULT;
  19. 	run->exit_reason = KVM_EXIT_UNKNOWN;
  20. 	switch (trap->scause) {
  21. 	/* 虚拟指令异常 */
  22. 	case EXC_VIRTUAL_INST_FAULT:    // 22, Virtual instruction
  23. 		if (vcpu->arch.guest_context.hstatus & HSTATUS_SPV)
  24. 			ret = kvm_riscv_vcpu_virtual_insn(vcpu, run, trap);
  25. 		break;
  26. 	/* 虚拟机内页缺陷异常 */
  27. 	case EXC_INST_GUEST_PAGE_FAULT: // 20, Instruction guest-page fault
  28. 	case EXC_LOAD_GUEST_PAGE_FAULT: // 21, Load guest-page fault
  29. 	case EXC_STORE_GUEST_PAGE_FAULT:// 23, Store/AMO guest-page fault
  30. 		if (vcpu->arch.guest_context.hstatus & HSTATUS_SPV)
  31. 			ret = gstage_page_fault(vcpu, run, trap);
  32. 		break;
  33. 	/* 虚拟机内系统调用 */
  34. 	case EXC_SUPERVISOR_SYSCALL:    // 10, Environment call from VS-mode
  35. 		if (vcpu->arch.guest_context.hstatus & HSTATUS_SPV)
  36. 			ret = kvm_riscv_vcpu_sbi_ecall(vcpu, run);
  37. 		break;
  38. 	default:
  39. 		break;
  40. 	}
  42. 	/* 若异常未能被顺利处理（ret > 0），则输出当前状态（sepc, sstatus, hstatus）和对应的异常信息（scause, stval, htval, htinst）*/
  43. 	/* Print details in-case of error */
  44. 	if (ret < 0) {
  45. 		kvm_err("VCPU exit error %d\n", ret);
  46. 		kvm_err("SEPC=0x%lx SSTATUS=0x%lx HSTATUS=0x%lx\n",
  47. 			vcpu->arch.guest_context.sepc,
  48. 			vcpu->arch.guest_context.sstatus,
  49. 			vcpu->arch.guest_context.hstatus);
  50. 		kvm_err("SCAUSE=0x%lx STVAL=0x%lx HTVAL=0x%lx HTINST=0x%lx\n",
  51. 			trap->scause, trap->stval, trap->htval, trap->htinst);
  52. 	}
  54. 	return ret;
  55. }

如上所示，KVM 的实现中包含了三类异常：

  • 虚拟指令异常；
  • Guest page fault；
  • SBI 系统调用。

特权指令集手册 中规定了每种异常对应的编码（即 scause 的可能的值），在进行异常处理时，可依据据 scause 的具体值确定其处理方式，如下表所示。

在 KVM 中，其对应宏的定义如下：

  1. // arch/riscv/include/asm/csr.h: line 66
  2. /* Exception causes */
  3. #define EXC_INST_MISALIGNED	0               // Instruction address misaligned
  4. #define EXC_INST_ACCESS		1               // Instruction access fault
  5. #define EXC_INST_ILLEGAL	2               // Illegal instruction
  6. #define EXC_BREAKPOINT		3               // Breakpoint
  7. #define EXC_LOAD_ACCESS		5               // Load access fault
  8. #define EXC_STORE_ACCESS	7               // Store/AMO access fault
  9. #define EXC_SYSCALL		8		// Environment call from U-mode or VU-mode
  10. #define EXC_HYPERVISOR_SYSCALL	9	        // Environment call from HS-mode
  11. #define EXC_SUPERVISOR_SYSCALL	10		// Environment call from VS-mode
  12. #define EXC_INST_PAGE_FAULT	12		// Instruction page fault
  13. #define EXC_LOAD_PAGE_FAULT	13		// Load page fault
  14. #define EXC_STORE_PAGE_FAULT	15		// Store/AMO page fault
  15. #define EXC_INST_GUEST_PAGE_FAULT	20      // Instruction guest-page fault
  16. #define EXC_LOAD_GUEST_PAGE_FAULT	21      // Load guest-page fault
  17. #define EXC_VIRTUAL_INST_FAULT		22      // Virtual instruction
  18. #define EXC_STORE_GUEST_PAGE_FAULT	23      // Store/AMO guest-page fault

3.3 虚拟指令异常

其中，EXC_VIRTUAL_INST_FAULT 即 virtual instruction exception 对应如下情况：

  • 在 VS-Mode 或 VU-Mode 下访问特定 CSR 的特定位；
  • 在 VS-Mode 或 VU-Mode 下执行无权限的指令如 HFENCE, HLV, HSV 等。

KVM 中 virtual instruction 异常的处理如下：

  1. // arch/riscv/kvm/vcpu_insn.c: line 397
  2. /**
  3.  * kvm_riscv_vcpu_virtual_insn -- Handle virtual instruction trap
  4.  *
  5.  * @vcpu: The VCPU pointer
  6.  * @run:  The VCPU run struct containing the mmio data
  7.  * @trap: Trap details
  8.  *
  9.  * Returns > 0 to continue run-loop
  10.  * Returns   0 to exit run-loop and handle in user-space.
  11.  * Returns < 0 to report failure and exit run-loop
  12.  */
  13. int kvm_riscv_vcpu_virtual_insn(struct kvm_vcpu *vcpu, struct kvm_run *run,
  14. 				struct kvm_cpu_trap *trap)
  15. {
  16. 	unsigned long insn = trap->stval;   // 获取导致 trap 的指令
  17. 	struct kvm_cpu_trap utrap = { 0 };
  18. 	struct kvm_cpu_context *ct;
  20. 	/* 判断是否为 16-bit 的压缩指令（[1:0]=2，非压缩指令 [1:0]=3），如果是压缩指令，则作如下处理 */
  21. 	if (unlikely(INSN_IS_16BIT(insn))) {
  22. 		if (insn == 0) {    // Illegal instruction 非法指令（参见特权指令集手册的表 16.5）
  23. 			ct = &vcpu->arch.guest_context;
  24. 			insn = kvm_riscv_vcpu_unpriv_read(vcpu, true,
  25. 							  ct->sepc,
  26. 							  &utrap);  // 从 Guest 内存中读取指定地址的内存
  27. 			if (utrap.scause) {
  28. 				utrap.sepc = ct->sepc;
  29. 				kvm_riscv_vcpu_trap_redirect(vcpu, &utrap); // 重定向 trap 到 Guest 中
  30. 				return 1;
  31. 			}
  32. 		}
  33. 		if (INSN_IS_16BIT(insn))
  34. 			return truly_illegal_insn(vcpu, run, insn);     // 将当前指令直接重定向到 Guest 中
  35. 	}
  37. 	/* 对于非压缩指令，根据当前指令的类型（[6:2] opcode）进行处理：*/
  38. 	switch ((insn & INSN_OPCODE_MASK) >> INSN_OPCODE_SHIFT) {
  39. 	case INSN_OPCODE_SYSTEM:    // SYSTEM 类型的指令（ecall, ebreak, CSR 读写指令）
  40. 		return system_opcode_insn(vcpu, run, insn);
  41. 	default:    // 正常长度且非 SYSTEM 类型的指令，将当前指令直接重定向到 Guest 中进行处理
  42. 		return truly_illegal_insn(vcpu, run, insn);
  43. 	}
  44. }
  46. // arch/riscv/kvm/vcpu_insn.c: line 70-72
  47. #define INSN_16BIT_MASK		0x3
  48. #define INSN_IS_16BIT(insn)	(((insn) & INSN_16BIT_MASK) != INSN_16BIT_MASK)

处理非法压缩指令时，用于从 Guest 获取合法指令的 kvm_riscv_vcpu_unpriv_read 函数：

  2. /**

  6.  * @read_insn: Flag representing whether we are reading instruction
  7.  * @guest_addr: Guest address to read
  8.  * @trap: Output pointer to trap details
  9.  */

从 Hypervisor 重定向到 Guest 的函数 kvm_riscv_vcpu_trap_redirect：

  2. /**
  3.  * kvm_riscv_vcpu_trap_redirect -- Redirect trap to Guest
  4.  *
  5.  * @vcpu: The VCPU pointer
  6.  * @trap: Trap details
  7.  */
  8. void kvm_riscv_vcpu_trap_redirect(struct kvm_vcpu *vcpu,
  9. 				  struct kvm_cpu_trap *trap);

对合法的压缩指令以及非 SYSTEM 类型的非压缩指令，不进行额外处理，直接调用 truly_illegal_insn 函数处理，保存当前 trap 的具体信息，将 Guest PC 设置为 Guest 中对应的异常向量， 然后返回到到 Guest 中对异常进行处理：

  1. // arch/riscv/kvm/vcpu_insn.c: line 151
  2. static int truly_illegal_insn(struct kvm_vcpu *vcpu, struct kvm_run *run,
  3. 			      ulong insn)
  4. {
  5. 	struct kvm_cpu_trap utrap = { 0 };
  7. 	/* Redirect trap to Guest VCPU */
  8. 	utrap.sepc = vcpu->arch.guest_context.sepc;
  9. 	utrap.scause = EXC_INST_ILLEGAL;
  10. 	utrap.stval = insn;
  11. 	utrap.htval = 0;
  12. 	utrap.htinst = 0;
  13. 	kvm_riscv_vcpu_trap_redirect(vcpu, &utrap);
  15. 	return 1;
  16. }

对 SYSTEM 类型进行特别操作的函数 system_opcode_insn：

  1. // arch/riscv/kvm/vcpu_insn.c: line 368
  2. static int system_opcode_insn(struct kvm_vcpu *vcpu, struct kvm_run *run,
  3. 			      ulong insn)
  4. {
  5. 	int i, rc = KVM_INSN_ILLEGAL_TRAP;
  6. 	const struct insn_func *ifn;
  8. 	for (i = 0; i < ARRAY_SIZE(system_opcode_funcs); i++) {
  9. 		ifn = &system_opcode_funcs[i];
  10. 		if ((insn & ifn->mask) == ifn->match) {
  11. 			rc = ifn->func(vcpu, run, insn);
  12. 			break;
  13. 		}
  14. 	}
  16. 	switch (rc) {
  17. 	case KVM_INSN_ILLEGAL_TRAP:         // 非法指令异常，设置 scause 为对应编码后重定向到 Guest
  19. 		return truly_illegal_insn(vcpu, run, insn);
  20. 	case KVM_INSN_VIRTUAL_TRAP:         // 虚拟指令异常，设置 scause 为对应编码后重定向到 Guest
  21. 		return truly_virtual_insn(vcpu, run, insn);
  22. 	case KVM_INSN_CONTINUE_NEXT_SEPC:   // 执行下一条指令
  23. 		vcpu->arch.guest_context.sepc += INSN_LEN(insn);
  24. 		break;
  25. 	default:
  26. 		break;
  27. 	}
  29. 	return (rc <= 0) ? rc : 1;
  30. }

虚拟指令的处理函数 truly_virtual_insn：

  1. // arch/riscv/kvm/vcpu_insn.c: line 167
  2. static int truly_virtual_insn(struct kvm_vcpu *vcpu, struct kvm_run *run,
  3. 			      ulong insn)
  4. {
  5. 	struct kvm_cpu_trap utrap = { 0 };
  7. 	/* Redirect trap to Guest VCPU */
  8. 	utrap.sepc = vcpu->arch.guest_context.sepc;
  9. 	utrap.scause = EXC_VIRTUAL_INST_FAULT;
  10. 	utrap.stval = insn;
  11. 	utrap.htval = 0;
  12. 	utrap.htinst = 0;
  13. 	kvm_riscv_vcpu_trap_redirect(vcpu, &utrap);
  15. 	return 1;
  16. }

3.5 SBI 系统调用

系统调用的处理通过调用 kvm_riscv_vcpu_sbi_ecall 函数实现，如下方代码块所示：

SBI（Supervisor Binary Interface） 是直接运行在 Machine Mode 下的，为上层 OS 提供统一接口的程序，具有最高权限。而 Guest 访问 SBI 系统调用，是在 KVM 中模拟实现，不是实际访问 Machine Mode 中的 SBI firmware。KVM 通过直接访问和设置寄存器（cp->a7, cp->a0, cp->a0 等）的值来实现对 SBI 系统调用的处理。

  1. int kvm_riscv_vcpu_sbi_ecall(struct kvm_vcpu *vcpu, struct kvm_run *run)
  2. {
  3. 	int ret = 1;
  4. 	bool next_sepc = true;
  5. 	bool userspace_exit = false;
  6. 	struct kvm_cpu_context *cp = &vcpu->arch.guest_context;
  7. 	const struct kvm_vcpu_sbi_extension *sbi_ext;
  8. 	struct kvm_cpu_trap utrap = { 0 };
  9. 	unsigned long out_val = 0;
  10. 	bool ext_is_v01 = false;
  12. 	/* 根据当前参数（a7）确定对应 SBI 中被调用的扩展（ext）及其 handler，获得返回值 ret；否则设置 ret 的值为不支持 SBI */
  13. 	sbi_ext = kvm_vcpu_sbi_find_ext(cp->a7);
  14. 	if (sbi_ext && sbi_ext->handler) {
  15. #ifdef CONFIG_RISCV_SBI_V01
  16. 		if (cp->a7 >= SBI_EXT_0_1_SET_TIMER &&
  17. 		    cp->a7 <= SBI_EXT_0_1_SHUTDOWN)
  18. 			ext_is_v01 = true;
  19. #endif
  20. 		ret = sbi_ext->handler(vcpu, run, &out_val, &utrap, &userspace_exit);
  21. 	} else {
  22. 		/* Return error for unsupported SBI calls */
  23. 		cp->a0 = SBI_ERR_NOT_SUPPORTED;
  24. 		goto ecall_done;
  25. 	}
  27. 	/* 依据经由 SBI ext handler 处理之后返回的 utrap 判断是否为需要进一步处理的 trap 等 */
  28. 	/* Handle special error cases i.e trap, exit or userspace forward */
  29. 	if (utrap.scause) {
  30. 		/* No need to increment sepc or exit ioctl loop */
  31. 		ret = 1;
  32. 		utrap.sepc = cp->sepc;
  33. 		kvm_riscv_vcpu_trap_redirect(vcpu, &utrap);
  34. 		next_sepc = false;
  35. 		goto ecall_done;
  36. 	}
  38. 	/* 依据 SBI 返回结果判断是否需要停止运行，或直接向 Guest 返回特定错误代码 */
  39. 	/* Exit ioctl loop or Propagate the error code the guest */
  40. 	if (userspace_exit) {
  41. 		next_sepc = false;
  42. 		ret = 0;
  43. 	} else {
  44. 		/**
  45. 		 * SBI extension handler always returns an Linux error code. Convert
  46. 		 * it to the SBI specific error code that can be propagated the SBI
  47. 		 * caller.
  48. 		 */
  49. 		ret = kvm_linux_err_map_sbi(ret);
  50. 		cp->a0 = ret;
  51. 		ret = 1;
  52. 	}
  53. /* 设置全局 pc 以及返回值 */
  54. ecall_done:
  55. 	if (next_sepc)
  56. 		cp->sepc += 4;
  57. 	if (!ext_is_v01)
  58. 		cp->a1 = out_val;
  60. 	return ret;
  61. }
